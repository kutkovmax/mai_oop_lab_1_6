cmake_minimum_required(VERSION 3.10)
project(lab1-6)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(GTest REQUIRED)

add_library(remove_wovels src/remove_wovels.cpp)

add_executable(main src/main.cpp)
target_link_libraries(main PRIVATE remove_wovels)

# Тесты
enable_testing()

add_executable(test_remove_wovels tests/test_remove_wovels.cpp)
target_link_libraries(test_remove_wovels PRIVATE remove_wovels GTest::GTest GTest::Main)

add_test(NAME test_remove_wovels COMMAND test_remove_wovels)
